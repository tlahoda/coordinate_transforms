 22 /**
 23  * Precalculated pi / 2.
 24  */
 25 Math.piOverTwo = Math.PI / 2.0;
 26 
 27 /**
 28  * Precalculated deg2rad conversion factor.
 29  */
 30 Math.deg2rad = Math.PI / 180.0;
 31 
 32 /**
 33  * Converts a lat/lon point to a two dimensional point in an orthographic projection.
 34  *
 35  * @param point The lat/lon point.
 36  *
 37  * @return The two dimensional point.
 38  */
 39 function toOrtho (point) {
 40   point[0] *= Math.deg2rad;
 41   point[1] *= Math.deg2rad;
 42   var slat = Math.sin (Math.piOverTwo - point[0]);
 43   return [slat * Math.cos (point[1]), slat * Math.sin (point[1])];
 44 }
 45 
 46 /**
 47  * Converts a lat/lon point to a three dimensional point.
 48  *
 49  * @param point The lat/lon point.
 50  *
 51  * @return The WebGL dimensional point.
 52  */
 53 function toWebGL (point) {
 54   point[0] *= Math.deg2rad;
 55   point[1] *= Math.deg2rad;
 56   var clat = Math.cos (point[0]);
 57   return [clat * Math.cos (point[1]), Math.sin (point[0]), -(clat * Math.sin (point[1]))];
 58 }

 60 /**
 61  * Scales a vertex.
 62  *
 63  * @param vertex The vertex.
 64  * @param s The scaling factor.
 65  * @param axis The axis to scale. If not present all axes are scaled.
 66  *
 67  * @return The scaled vertex.
 68  */
 69 function scale (vertex, s, axis) {
 70   if (axis == undefined)
 71     return vertex.apply (function (ele, s) { return ele * s; }, s);
 72   vertex[axis] *= s;
 73   return vertex;
 74 }
 75 
 76 /**
 77  * Shifts a vertex.
 78  *
 79  * @param vertex The vertex.
 80  * @param s The amount to shift.
 81  * @param axis The axis to shift. If not present all axes are shifted.
 82  *
 83  * @return The shifted vertex.
 84  */
 85 function shift (vertex, s, axis) {
 86   if (axis == undefined)
 87     return vertex.apply (function (ele, s) { return ele + s; }, s);
 88   vertex[axis] += s;
 89   return vertex;
 90 }
 91 
 92 /**
 93  * Inverts a vertex.
 94  *
 95  * @param vertex The vertex.
 96  * @param bound The max bound of the axis.
 97  * @param axis The axis to invert. If not present all axes are inverted.
 98  *
 99  * @return The inverted vertex.
100  */
101 function invert (vertex, bound, axis) {
102   if (axis == undefined)
103     return vertex.apply (function (ele, bound) { return bound - ele; }, bound);
104   vertex[axis] = bound - vertex[axis];
105   return vertex;
106 }
